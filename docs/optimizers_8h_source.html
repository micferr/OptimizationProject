466 bool hill_climbing(optimization_problem& problem) {
467  require_single_objective(problem);
468  require_integer(problem);
469  // For a constrained problem, a feasible starting assignment is required
470  if (problem.is_constrained()) {
471  require_feasible(problem);
472  }
473 
474  auto& params = problem.get_params();
475  while (true) {
476  // 0 -> first parameter decreased by one
477  // 1 -> first parameter increased by one
478  // 2 -> second parameter decreased by one
479  // ...
480  // -1 -> no feasible neighbour
481  int best_neighbour_id = -1;
482  float best_fitness = problem.get_objective_value(params);
483 
484  // Find the best neighbour
485  for (auto i = 0; i < params.size(); i++) {
486  auto& p = params[i];
487  auto v = p.get_ival();
488 
489  // Decrease parameter by one
490  if (p.get_imin() < v) {
491  p.set_ival(v - 1);
492  if (problem.is_feasible(params)) {
493  auto f = problem.get_objective_value(params);
494  if (f > best_fitness) {
495  best_fitness = f;
496  best_neighbour_id = 2 * i;
497  }
498  }
499  p.set_ival(v);
500  }
501 
502  // Increase parameter by one
503  if (v < p.get_imax()) {
504  p.set_ival(v + 1);
505  if (problem.is_feasible(params)) {
506  auto f = problem.get_objective_value(params);
507  if (f > best_fitness) {
508  best_fitness = f;
509  best_neighbour_id = 2 * i + 1;
510  }
511  }
512  p.set_ival(v);
513  }
514  }
515  if (best_neighbour_id != -1) {
516  auto& param = params[best_neighbour_id / 2];
517  auto v = param.get_ival();
518  auto delta = best_neighbour_id % 2 ? +1 : -1;
519  param.set_ival(v + delta);
520  }

573  p.set_rval(v);
574  }
575

577  if (v + delta < p.get_rmax()) {
578  p.set_rval(v + delta);
579  if (problem.is_feasible(params)) {
580  auto f = problem.get_objective_value(params);
581  if (f > best_fitness) {
582  best_fitness = f;
583  best_neighbour_id = 2 * i + 1;
584  }
585  }
586  p.set_rval(v);
587  }
588  }
589 
590  if (best_neighbour_id != -1) { // Better neighbour found
591  auto& param = params[best_neighbour_id / 2];
592  param.set_rval(param.get_rval() + (best_neighbour_id % 2 ? delta : -delta));
593  }
594  else if (delta/2.f >= eps) { // delta can be halved
595  delta /= 2.f;
596  }

933  // Truncation operation
934 
935  // Less-than operator described in the paper (section 3.2)
936  //
937  // i1 is less than i2 in the weak ordering iff either:
938  // - sigma_i_k = sigma_j_k for all k
939  // - Exists k s.t. sigma_i_l = sigma_j_l for 0 < l < k and sigma_i_k < sigma_j_k
940  auto na_copy = next_archive;
941  auto less_than = [&](const auto& i1, const auto& i2) {
942  std::vector<individual*> dist1, dist2;
943  for (auto& i : na_copy) {
944  dist1.push_back(&i);
945  dist2.push_back(&i);
946  }

949  std::sort(dist1.begin(), dist1.end(), [&](auto& _i1, auto& _i2) {
950  return
951  squared_distance(i1.params, _i1->params) <
952  squared_distance(i1.params, _i2->params);
953  });
954  std::sort(dist2.begin(), dist2.end(), [&](auto& _i1, auto& _i2) {
955  return
956  squared_distance(i2.params, _i1->params) <
957  squared_distance(i2.params, _i2->params);
958  });
959 
960  // Ignore the nearest element as it's the individual itself
961  for (auto i = 1; i < dist1.size(); i++) {
962  auto sigma_i = squared_distance(i1.params, dist1[i]->params);
963  auto sigma_j = squared_distance(i2.params, dist2[i]->params);
964  if (sigma_i < sigma_j) return true;
965  else if (sigma_i > sigma_j) return false;
966  }
967  return false;
968  };
969  // Find the minimum element and delete it
970  std::sort(next_archive.begin(), next_archive.end(), less_than);
971  next_archive.erase(next_archive.begin());
972  std::shuffle(next_archive.begin(), next_archive.end(), rg.rand_engine);
973  }

975  // Choose the best dominated individuals
976  std::sort(dominated.begin(), dominated.end(), [&](auto i1, auto i2) {
977  return
978  raw_fitness(*i1, population, archive) + density(*i1, population, archive) <
979  raw_fitness(*i2, population, archive) + density(*i2, population, archive);
980  });
981  auto num_needed = archive_size - next_archive.size();
982  for (auto i = 0; i < num_needed; i++) {
983  next_archive.push_back(*(dominated[i]));
984  }
985  }

988  population.clear();
989  for (auto i = 0; i < population_size; i++) {
990  unsigned parents[2];
991  for (auto j = 0; j < 2; j++) {
992  // Choose two random individual for each parent role, and choose the best
993  // among the two with probability 0.7
994  unsigned candidate_parents[2];
995  for (auto k = 0; k < 2; k++) candidate_parents[k] = rg.rand(0, next_archive.size()-1);
996  auto fitness0 = raw_fitness(next_archive[candidate_parents[0]], {}, next_archive);
997  auto fitness1 = raw_fitness(next_archive[candidate_parents[1]], {}, next_archive);
998  if (fitness0 > fitness1) {
999  std::swap(candidate_parents[0], candidate_parents[1]);
1000  }
1001  parents[j] = candidate_parents[rg.rand(0.f, 1.f) < 0.7f ? 0 : 1];
1002  }
1003 
1004  individual child;
1005  child.params = mating_operator(
1006  next_archive[parents[0]].params, next_archive[parents[1]].params
1007  );
1008  mutation_operator(child.params);
1009  child.objective_values = problem.get_objectives_values(child.params);
1010  population.push_back(child);
1011  }
1012  archive = next_archive;
1013  }

1016  // Take all non-dominated individuals from the archive as solutions

1018  bool dominated = false;

1021  if (archive[j].dominates(archive[i])) {

1024  }
1025  }
